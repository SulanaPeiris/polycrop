// ================= TIME (TLS) =================
bool initTime() {
  configTime(19800, 0, "pool.ntp.org", "time.nist.gov"); // Sri Lanka UTC+5:30
  time_t now = time(nullptr);
  int retry = 0;
  while (now < 1700000000 && retry < 60) {
    delay(500);
    now = time(nullptr);
    retry++;
  }
  return now >= 1700000000;
}

String nowRFC3339Z() {
  time_t now = time(nullptr);
  if (now < 1700000000) return "1970-01-01T00:00:00Z";

  struct tm tm_utc;
  gmtime_r(&now, &tm_utc);

  char buf[25];
  strftime(buf, sizeof(buf), "%Y-%m-%dT%H:%M:%SZ", &tm_utc);
  return String(buf);
}

  }

  fbdo.setResponseSize(4096);

  delay(700);

// ================= LORA =================
bool initLoRaWithOLED() {
  oledStatus("LoRa: Initializing...", "433MHz SX1278");

  SPI.begin(LORA_SCK, LORA_MISO, LORA_MOSI, LORA_CS);
  LoRa.setPins(LORA_CS, LORA_RST, LORA_DIO0);

  if (!LoRa.begin(LORA_FREQ)) {
    oledStatus("LoRa: INIT FAILED", "Check wiring", "3.3V + antenna");
    return false;
  }

  LoRa.setSpreadingFactor(7);
  LoRa.setSignalBandwidth(125E3);
  LoRa.setCodingRate4(5);

  oledStatus("LoRa: Receiver Ready");
  delay(700);
  return true;
}

// ================= PARSE =================
// incoming: counter,t1,h1,t2,h2
bool parseCSV5(const String &s, String &counterStr, float &t1, float &h1, float &t2, float &h2) {
  int p1 = s.indexOf(',');
  int p2 = s.indexOf(',', p1 + 1);
  int p3 = s.indexOf(',', p2 + 1);
  int p4 = s.indexOf(',', p3 + 1);
  if (p1 < 0 || p2 < 0 || p3 < 0 || p4 < 0) return false;

  counterStr = s.substring(0, p1);
  t1 = s.substring(p1 + 1, p2).toFloat();
  h1 = s.substring(p2 + 1, p3).toFloat();
  t2 = s.substring(p3 + 1, p4).toFloat();
  h2 = s.substring(p4 + 1).toFloat();
  return true;
}

String makeDocId() {
  uint32_t r = (uint32_t)esp_random();
  return String(GATEWAY_ID) + "_" + String(millis()) + "_" + String(r, HEX);
}

// ================= FETCH ASSIGNED TUNNEL =================
void fetchAssignedTunnelId() {
  if (!Firebase.ready()) return;

  String path = String(GATEWAYS_COLLECTION) + "/" + String(GATEWAY_ID);

  if (!Firebase.Firestore.getDocument(&fbdo, FIREBASE_PROJECT_ID, "", path.c_str())) {
    return;
  }

  FirebaseJson json;
  json.setJsonData(fbdo.payload().c_str());

  FirebaseJsonData data;
  json.get(data, "fields/assignedTunnelId/stringValue");
  if (data.success) {
    assignedTunnelId = data.to<String>();
  }
}

// ================= UPDATE GATEWAY DOC =================
void updateGatewayDoc() {
  if (!Firebase.ready()) return;

  String path = String(GATEWAYS_COLLECTION) + "/" + String(GATEWAY_ID);

  FirebaseJson content;
  content.set("fields/gatewayId/stringValue", String(GATEWAY_ID));
  content.set("fields/assignedTunnelId/stringValue", assignedTunnelId);
  content.set("fields/lastSeenAt/timestampValue", nowRFC3339Z());
  content.set("fields/lastRssi/integerValue", (int)lastRSSI);
  content.set("fields/lastSnr/doubleValue", isnan(lastSNR) ? 0.0 : lastSNR);

  String mask = "gatewayId,assignedTunnelId,lastSeenAt,lastRssi,lastSnr";

  Firebase.Firestore.patchDocument(&fbdo, FIREBASE_PROJECT_ID, "", path.c_str(), content.raw(), mask.c_str());
}

// ================= FIRESTORE WRITE =================
bool storeToFirestore() {
  if (!Firebase.ready()) return false;
  if (assignedTunnelId.length() == 0) return false;

  String readingPath = String("tunnels/") + assignedTunnelId + "/" + String(READINGS_SUBCOLLECTION) + "/" + makeDocId();
  String summaryPath = String("tunnels/") + assignedTunnelId + "/sensorSummary/latest";

  float avgT = (lastT1 + lastT2) / 2.0;
  float avgH = (lastH1 + lastH2) / 2.0;

  // 1) Create reading doc
  FirebaseJson reading;
  reading.set("fields/gatewayId/stringValue", String(GATEWAY_ID));
  reading.set("fields/counter/stringValue", lastCounter);

  reading.set("fields/s1_temp/doubleValue", lastT1);
  reading.set("fields/s1_hum/doubleValue", lastH1);
  reading.set("fields/s2_temp/doubleValue", lastT2);
  reading.set("fields/s2_hum/doubleValue", lastH2);

  reading.set("fields/avg_temp/doubleValue", avgT);
  reading.set("fields/avg_hum/doubleValue", avgH);

  reading.set("fields/rssi/integerValue", (int)lastRSSI);
  reading.set("fields/snr/doubleValue", isnan(lastSNR) ? 0.0 : lastSNR);
  reading.set("fields/ts/timestampValue", nowRFC3339Z());

  bool ok1 = Firebase.Firestore.createDocument(&fbdo, FIREBASE_PROJECT_ID, "", readingPath.c_str(), reading.raw());
  if (!ok1) {
    Serial.print("Reading create FAILED: ");
    Serial.println(fbdo.errorReason());
  }

  // 2) Patch summary/latest (fast UI)
  FirebaseJson summary;
  summary.set("fields/gatewayId/stringValue", String(GATEWAY_ID));
  summary.set("fields/counter/stringValue", lastCounter);

  summary.set("fields/s1_temp/doubleValue", lastT1);
  summary.set("fields/s2_temp/doubleValue", lastT2);
  summary.set("fields/avg_temp/doubleValue", avgT);

  summary.set("fields/s1_hum/doubleValue", lastH1);
  summary.set("fields/s2_hum/doubleValue", lastH2);
  summary.set("fields/avg_hum/doubleValue", avgH);

  summary.set("fields/rssi/integerValue", (int)lastRSSI);
  summary.set("fields/snr/doubleValue", isnan(lastSNR) ? 0.0 : lastSNR);
  summary.set("fields/lastSeenAt/timestampValue", nowRFC3339Z());

  String mask = "gatewayId,counter,s1_temp,s2_temp,avg_temp,s1_hum,s2_hum,avg_hum,rssi,snr,lastSeenAt";

  bool ok2 = Firebase.Firestore.patchDocument(&fbdo, FIREBASE_PROJECT_ID, "", summaryPath.c_str(), summary.raw(), mask.c_str());
  if (!ok2) {
    Serial.print("Summary patch FAILED: ");
    Serial.println(fbdo.errorReason());
  }

  updateGatewayDoc();
  return ok1 && ok2;
}

// ================= SETUP =================
void setup() {
  Serial.begin(115200);

  Wire.begin(I2C_SDA, I2C_SCL);
  if (!oledBegin()) {
    Serial.println("OLED not found at 0x3C/0x3D");
    while (1) delay(10);
  }

  oledStatus("Booting...");
  delay(500);

  connectWiFiWithOLED();

  oledStatus("Time: Syncing (NTP)...");
  bool timeOk = initTime();
  oledStatus(String("Time: ") + (timeOk ? "OK" : "FAILED"));
  delay(700);

  initFirebaseWithOLED();

  if (!initLoRaWithOLED()) {
    while (1) delay(10);
  }

  fetchAssignedTunnelId();
  lastAssignFetchMs = millis();

  lastPacketMs = millis();
  showWaitingScreen();

  Serial.println("Receiver ready.");
}

// ================= LOOP =================
void loop() {
  Firebase.ready(); // keep token alive

  // refresh assigned tunnel
  if (Firebase.ready() && (millis() - lastAssignFetchMs > ASSIGN_REFRESH_MS)) {
    fetchAssignedTunnelId();
    lastAssignFetchMs = millis();
  }

  int packetSize = LoRa.parsePacket();
  if (packetSize) {
    String incoming = "";
    while (LoRa.available()) incoming += (char)LoRa.read();

    int rssi = LoRa.packetRssi();
    float snr = LoRa.packetSnr(); // ✅ connectivity info

    String counterStr;
    float t1, h1, t2, h2;

    if (parseCSV5(incoming, counterStr, t1, h1, t2, h2)) {
      lastPacketMs = millis();

      lastCounter = counterStr;
      lastT1 = t1; lastH1 = h1;
      lastT2 = t2; lastH2 = h2;
      lastRSSI = rssi;
      lastSNR = snr;
      haveData = true;

      drawData(t1, h1, t2, h2, rssi, snr);
    }
  }

  if ((millis() - lastPacketMs) > WAITING_TIMEOUT_MS) {
    if (!showingWaiting) showWaitingScreen();
  }

  if (haveData && (millis() - lastStoreMs >= STORE_INTERVAL_MS)) {
    lastStoreMs = millis();

    if (WiFi.status() != WL_CONNECTED) {
      oledStatus("WiFi: Reconnecting...");
      connectWiFiWithOLED();
      oledStatus("Time: Syncing (NTP)...");
      initTime();
      fetchAssignedTunnelId();
      showWaitingScreen();
    }

    storeToFirestore();
  }
}
