// ================= TIME (TLS) =================
bool initTime() {
  configTime(19800, 0, "pool.ntp.org", "time.nist.gov"); // Sri Lanka UTC+5:30
  time_t now = time(nullptr);
  int retry = 0;
  while (now < 1700000000 && retry < 60) {
    delay(500);
    now = time(nullptr);
    retry++;
  }
  return (now >= 1700000000);
}

String nowRFC3339Z() {
  time_t now = time(nullptr);
  if (now < 1700000000) return "1970-01-01T00:00:00Z";
  struct tm tm_utc;
  gmtime_r(&now, &tm_utc);
  char buf[25];
  strftime(buf, sizeof(buf), "%Y-%m-%dT%H:%M:%SZ", &tm_utc);
  return String(buf);
}

// ================= WIFI =================
bool connectWiFiWithOLED() {
  oledStatus("WiFi: Connecting...", WIFI_SSID);

  WiFi.begin(WIFI_SSID, WIFI_PASSWORD);
  unsigned long start = millis();

  while (WiFi.status() != WL_CONNECTED) {
    delay(250);
    // timeout after 20s (still continue trying, but show message)
    if (millis() - start > 20000) {
      oledStatus("WiFi: Still trying...", "Check password/router");
      start = millis();
    }
  }

  WiFi.setSleep(false);

  String ip = WiFi.localIP().toString();
  oledStatus("WiFi: Connected", "IP: " + ip);
  delay(800);
  return true;
}

// ================= FIREBASE =================
bool initFirebaseWithOLED() {
  oledStatus("Firebase: Auth...", "Anonymous SignUp");

  config.api_key = API_KEY;
  config.token_status_callback = tokenStatusCallback;

  bool ok = Firebase.signUp(&config, &auth, "", "");
  if (!ok) {
    oledStatus("Firebase: SignUp FAIL", String(config.signer.signupError.message.c_str()));
    delay(2000);
    return false;
  }

  fbdo.setBSSLBufferSize(4096, 1024);
  fbdo.setResponseSize(4096);

  Firebase.begin(&config, &auth);
  Firebase.reconnectWiFi(true);

  // wait a moment for token
  unsigned long start = millis();
  while (!Firebase.ready() && millis() - start < 8000) {
    delay(200);
  }

  oledStatus("Firebase: " + String(Firebase.ready() ? "Ready" : "Not ready"));
  delay(800);
  return Firebase.ready();
}

// ================= LORA =================
bool initLoRaWithOLED() {
  oledStatus("LoRa: Initializing...", "433MHz SX1278");

  SPI.begin(LORA_SCK, LORA_MISO, LORA_MOSI, LORA_CS);
  LoRa.setPins(LORA_CS, LORA_RST, LORA_DIO0);

  if (!LoRa.begin(LORA_FREQ)) {
    oledStatus("LoRa: INIT FAILED", "Check wiring", "3.3V + antenna");
    return false;
  }

  // Must match TX
  LoRa.setSpreadingFactor(7);
  LoRa.setSignalBandwidth(125E3);
  LoRa.setCodingRate4(5);

  oledStatus("LoRa: Receiver Ready");
  delay(800);
  return true;
}

// ================= PARSE =================
// incoming: counter,t1,h1,t2,h2
bool parseCSV5(const String &s, String &counterStr, float &t1, float &h1, float &t2, float &h2) {
  int p1 = s.indexOf(',');
  int p2 = s.indexOf(',', p1 + 1);
  int p3 = s.indexOf(',', p2 + 1);
  int p4 = s.indexOf(',', p3 + 1);
  if (p1 < 0 || p2 < 0 || p3 < 0 || p4 < 0) return false;

  counterStr = s.substring(0, p1);
  t1 = s.substring(p1 + 1, p2).toFloat();
  h1 = s.substring(p2 + 1, p3).toFloat();
  t2 = s.substring(p3 + 1, p4).toFloat();
  h2 = s.substring(p4 + 1).toFloat();
  return true;
}

String makeDocId() {
  uint32_t r = (uint32_t)esp_random();
  return String(ROBOT_ID) + "_" + String(millis()) + "_" + String(r, HEX);
}

// ================= FIRESTORE WRITE =================
bool storeToFirestore() {
  if (!Firebase.ready()) return false;

  String documentPath = String(LORA_COLLECTION) + "/" + makeDocId();

  FirebaseJson content;
  content.set("fields/robotId/stringValue", String(ROBOT_ID));
  content.set("fields/counter/stringValue", lastCounter);

  content.set("fields/s1_temp/doubleValue", lastT1);
  content.set("fields/s1_hum/doubleValue", lastH1);
  content.set("fields/s2_temp/doubleValue", lastT2);
  content.set("fields/s2_hum/doubleValue", lastH2);

  content.set("fields/rssi/integerValue", String(lastRSSI));
  content.set("fields/ts/timestampValue", nowRFC3339Z());

  bool ok = Firebase.Firestore.createDocument(&fbdo, FIREBASE_PROJECT_ID, "",
                                             documentPath.c_str(), content.raw());

  if (!ok) {
    Serial.print("Firestore FAILED: ");
    Serial.println(fbdo.errorReason());
  }
  return ok;
}

// ================= SETUP =================
void setup() {
  Serial.begin(115200);

  Wire.begin(I2C_SDA, I2C_SCL);
  if (!oledBegin()) {
    Serial.println("OLED not found at 0x3C/0x3D");
    while (1) delay(10);
  }

  oledStatus("Booting...");
  delay(500);

  // WiFi
  connectWiFiWithOLED();

  // Time (TLS)
  oledStatus("Time: Syncing (NTP)...");
  bool timeOk = initTime();
  oledStatus(String("Time: ") + (timeOk ? "OK" : "FAILED"));
  delay(800);

  // Firebase
  initFirebaseWithOLED();

  // LoRa receiver init
  if (!initLoRaWithOLED()) {
    while (1) delay(10);
  }

  // Show waiting screen
  lastPacketMs = millis();
  showWaitingScreen();

  Serial.println("Receiver ready.");
}

// ================= LOOP =================
void loop() {
  Firebase.ready(); // keep token alive

  // ---- Receive packet ----
  int packetSize = LoRa.parsePacket();
  if (packetSize) {
    String incoming = "";
    while (LoRa.available()) incoming += (char)LoRa.read();

    int rssi = LoRa.packetRssi();

    Serial.print("Received: ");
    Serial.print(incoming);
    Serial.print(" | RSSI: ");
    Serial.println(rssi);

    String counterStr;
    float t1, h1, t2, h2;

    if (parseCSV5(incoming, counterStr, t1, h1, t2, h2)) {
      lastPacketMs = millis();

      lastCounter = counterStr;
      lastT1 = t1; lastH1 = h1;
      lastT2 = t2; lastH2 = h2;
      lastRSSI = rssi;
      haveData = true;

      drawData(t1, h1, t2, h2, rssi);
    }
  }

  // ---- If no packet for a while, show waiting ----
  if ((millis() - lastPacketMs) > WAITING_TIMEOUT_MS) {
    if (!showingWaiting) showWaitingScreen();
  }

  // ---- Store every 5 seconds ----
  if (haveData && (millis() - lastStoreMs >= STORE_INTERVAL_MS)) {
    lastStoreMs = millis();

    if (WiFi.status() != WL_CONNECTED) {
      oledStatus("WiFi: Reconnecting...");
      connectWiFiWithOLED();
      oledStatus("Time: Syncing (NTP)...");
      initTime();
      oledStatus("Time: OK");
      delay(300);
      showWaitingScreen();
    }

    storeToFirestore();
  }
}
