}

// ================= TIME SYNC (CRITICAL FOR TLS) =================
void initTime() {
  // Sri Lanka UTC +5:30 => 19800 seconds
  configTime(19800, 0, "pool.ntp.org", "time.nist.gov");

  Serial.print("Syncing time");
  time_t now = time(nullptr);
  int retry = 0;

  while (now < 1700000000 && retry < 60) { // wait until time is valid (>= ~2023)
    Serial.print(".");
    delay(500);
    now = time(nullptr);
    retry++;
  }

  Serial.println();
  if (now < 1700000000) {
    Serial.println("⚠️ Time sync failed. TLS may fail.");
  } else {
    Serial.print("Time OK: ");
    Serial.println(ctime(&now));
  }
}

// ✅ Firestore timestamp must be RFC3339 with Z (UTC)
// Example: 2026-03-02T07:59:20Z
String nowRFC3339Z() {
  time_t now = time(nullptr);
  if (now < 1700000000) return "1970-01-01T00:00:00Z"; // fallback (should not happen if NTP ok)

  struct tm tm_utc;
  gmtime_r(&now, &tm_utc);

  char buf[25];
  // YYYY-MM-DDTHH:MM:SSZ => 20 chars + null
  strftime(buf, sizeof(buf), "%Y-%m-%dT%H:%M:%SZ", &tm_utc);
  return String(buf);
}

// === FIREBASE/SCAN HELPERS ===
String getRFIDTag() {
  String tagID = "";
  for (byte i = 0; i < mfrc522.uid.size; i++) {
    tagID += String(mfrc522.uid.uidByte[i] < 0x10 ? "0" : "");
    tagID += String(mfrc522.uid.uidByte[i], HEX);
  }
  tagID.toUpperCase();
  return tagID;
}

// Create a reasonably-unique Firestore doc id for scanEvents
String makeEventDocId() {
  uint32_t r = (uint32_t)esp_random();
  String id = String(millis()) + "_" + String(r, HEX);
  return id;
}

  Serial.begin(115200);

  // === WIFI ===

  }
  Serial.println("\nConnected to WiFi!");

  // ✅ keep WiFi awake for stable TLS
  WiFi.setSleep(false);

  // ✅ sync time for TLS certificates
  initTime();

  // === FIREBASE INIT ===
  config.api_key = API_KEY;
  config.token_status_callback = tokenStatusCallback;

  // Anonymous sign-in
  if (Firebase.signUp(&config, &auth, "", "")) {
    Serial.println("Firebase Auth Ready");
  } else {
    Serial.print("Signup error: ");
    Serial.println(config.signer.signupError.message.c_str());
  }

  // ✅ bigger SSL buffers (reduces connection lost / write error)
  fbdo.setBSSLBufferSize(4096, 1024);  // RX, TX
  fbdo.setResponseSize(4096);

  Firebase.begin(&config, &auth);
  Firebase.reconnectWiFi(true);

  // Push initial default states to Firestore
  pushDataToFirebase();
  logScanEvent("BOOT", "Robot booted and connected to WiFi/Firebase");

  // LED and Pump setup
  pinMode(LED_PIN, OUTPUT);
  digitalWrite(LED_PIN, LOW);

  pinMode(PUMP_PIN, OUTPUT);
  digitalWrite(PUMP_PIN, LOW); // Ensure pump is OFF at startup

  // Motor Enable Pins
  pinMode(L_REN, OUTPUT);
  pinMode(L_LEN, OUTPUT);
  pinMode(R_REN, OUTPUT);
  pinMode(R_LEN, OUTPUT);

  digitalWrite(L_REN, HIGH);
  digitalWrite(L_LEN, HIGH);
  digitalWrite(R_REN, HIGH);
  digitalWrite(R_LEN, HIGH);

  // PWM Setup
  ledcSetup(CH_L_RPWM, PWM_FREQ, PWM_RES);
  ledcSetup(CH_L_LPWM, PWM_FREQ, PWM_RES);
  ledcSetup(CH_R_RPWM, PWM_FREQ, PWM_RES);
  ledcSetup(CH_R_LPWM, PWM_FREQ, PWM_RES);

  ledcAttachPin(L_RPWM, CH_L_RPWM);
  ledcAttachPin(L_LPWM, CH_L_LPWM);
  ledcAttachPin(R_RPWM, CH_R_RPWM);
  ledcAttachPin(R_LPWM, CH_R_LPWM);

  // IR Sensors
  pinMode(IR_LEFT, INPUT);
  pinMode(IR_RIGHT, INPUT);

  // RFID
  SPI.begin();        // ESP32 default VSPI pins
  mfrc522.PCD_Init();
  Serial.println("RFID Ready");

  // Stepper pins
  pinMode(EN_PIN, OUTPUT);
  digitalWrite(EN_PIN, LOW); // enable A4988 (LOW = enabled)

  stepper.setMaxSpeed(STEPPER_MAX_SPEED);
  stepper.setAcceleration(STEPPER_ACCEL);
  stepper.setCurrentPosition(0);

  stopMotors();
}

// ================= LOOP =================
void loop() {

  // --------- MODE: LINE FOLLOW ---------
  if (mode == LINE_FOLLOW) {
    modeStr = "LINE_FOLLOW";

    // RFID check (trigger stepper sequence)
    if (mfrc522.PICC_IsNewCardPresent() && mfrc522.PICC_ReadCardSerial()) {
      Serial.println("RFID detected -> STOP and rotate stepper CW 55 rounds");

      stopMotors();

      // === UPDATE STATE ON RFID DETECT ===
      currentRFID = getRFIDTag();
      robotMovingStatus = "STOPPED";
      scanningStatus = "SCANNING";
      pumpStatus = "OFF";

      // Write robot + plant + event
      pushDataToFirebase();
      updatePlantDocOnRFID();
      logScanEvent("RFID_DETECTED", "Robot stopped, scanning started");

      // Blink LED when RFID detected
      blinkLED(1, 150, 150);

      // Start CW move
      stepper.moveTo(TARGET_STEPS); // go to +TARGET_STEPS
      mode = STEPPER_CW;
      modeStr = "STEPPER_CW";

      logScanEvent("STEPPER_CW_START", "Stepper CW move started (55 rounds)");

      mfrc522.PICC_HaltA();
      mfrc522.PCD_StopCrypto1();
      return;
    }

    // Normal line follow
    int leftIR = digitalRead(IR_LEFT);
    int rightIR = digitalRead(IR_RIGHT);

    if (leftIR == LOW && rightIR == LOW) forward();
    else if (leftIR == LOW && rightIR == HIGH) leftTurn();
    else if (leftIR == HIGH && rightIR == LOW) rightTurn();
    else stopMotors();

    delay(10);
    return;
  }

  // --------- MODE: STEPPER CW ---------
  if (mode == STEPPER_CW) {
    modeStr = "STEPPER_CW";

    stopMotors();
    stepper.run();

    if (stepper.distanceToGo() == 0) {
      Serial.println("CW done -> rotate CCW 55 rounds");

      logScanEvent("STEPPER_CW_DONE", "Stepper CW finished, starting CCW");

      stepper.moveTo(0);
      mode = STEPPER_CCW;
      modeStr = "STEPPER_CCW";

      logScanEvent("STEPPER_CCW_START", "Stepper CCW move started (55 rounds)");
    }
    return;
  }

  // --------- MODE: STEPPER CCW ---------
  if (mode == STEPPER_CCW) {
    modeStr = "STEPPER_CCW";

    stopMotors();
    stepper.run();

    if (stepper.distanceToGo() == 0) {
      Serial.println("CCW done -> Turning ON water pump for 5 seconds");

      logScanEvent("STEPPER_CCW_DONE", "Stepper CCW finished, pump sequence starting");

      // === PUMP IS ON ===
      pumpStatus = "ON";
      scanningStatus = "IDLE";
      pushDataToFirebase();
      logScanEvent("PUMP_ON", "Pump turned ON");

      digitalWrite(PUMP_PIN, HIGH);
      delay(5000);
      digitalWrite(PUMP_PIN, LOW);

      Serial.println("Pump done -> resume line follow");

      // === RESUME MOVING ===
      pumpStatus = "OFF";
      robotMovingStatus = "MOVING";
      pushDataToFirebase();
      logScanEvent("PUMP_OFF", "Pump turned OFF, robot resuming line follow");

      updatePlantDocAfterPump();
      logScanEvent("PLANT_SCAN_DONE", "Plant scan marked DONE (pump finished)");

      mode = LINE_FOLLOW;
      modeStr = "LINE_FOLLOW";
    }
    return;
  }
}
