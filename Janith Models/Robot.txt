#include <SPI.h>
#include <MFRC522.h>
#include <AccelStepper.h>

// ================= MOTOR + IR PINS (UNCHANGED) =================
// Left Motor Driver
#define L_RPWM 25
#define L_LPWM 26
#define L_REN  27
#define L_LEN  14

// Right Motor Driver
#define R_RPWM 33
#define R_LPWM 32
#define R_REN  15
#define R_LEN  4

// IR Sensors
#define IR_LEFT  34
#define IR_RIGHT 35

// PWM Channels
#define CH_L_RPWM 0
#define CH_L_LPWM 1
#define CH_R_RPWM 2
#define CH_R_LPWM 3

#define PWM_FREQ 1000
#define PWM_RES 8
#define SPEED 180

// ================= RFID (RC522) =================
#define SS_PIN  5
#define RST_PIN 22
MFRC522 mfrc522(SS_PIN, RST_PIN);

// ================= A4988 STEPPER (NEW PINS) =================
#define STEP_PIN 16
#define DIR_PIN  17
#define EN_PIN   21

// AccelStepper: DRIVER mode uses STEP+DIR
AccelStepper stepper(AccelStepper::DRIVER, STEP_PIN, DIR_PIN);

// ================= LED (NEW) =================
// Most ESP32 dev boards have a built-in LED on GPIO 2.
// If yours is different, change this value.
#define LED_PIN 2

// ================= STEPPER SETTINGS =================
// If MS1=MS2=MS3 = HIGH => 1/16 microstep
const int MOTOR_STEPS_PER_REV = 200;  // 1.8° stepper = 200 full steps/rev

const long ROUNDS = 5;
const long TARGET_STEPS = ROUNDS * MOTOR_STEPS_PER_REV; // 15 revolutions

// Speed/accel (tune if needed)
const float STEPPER_MAX_SPEED = 2000;     // steps/sec
const float STEPPER_ACCEL     = 1500;     // steps/sec^2

// ================= STATE MACHINE =================
enum Mode {
  LINE_FOLLOW,
  STEPPER_CW,
  STEPPER_CCW
};

Mode mode = LINE_FOLLOW;

// ================= MOTOR FUNCTIONS =================
void forward() {
  ledcWrite(CH_L_RPWM, SPEED);
  ledcWrite(CH_L_LPWM, 0);
  ledcWrite(CH_R_RPWM, SPEED);
  ledcWrite(CH_R_LPWM, 0);
}

void leftTurn() {
  ledcWrite(CH_L_RPWM, 0);
  ledcWrite(CH_L_LPWM, 0);
  ledcWrite(CH_R_RPWM, SPEED);
  ledcWrite(CH_R_LPWM, 0);
}

void rightTurn() {
  ledcWrite(CH_L_RPWM, SPEED);
  ledcWrite(CH_L_LPWM, 0);
  ledcWrite(CH_R_RPWM, 0);
  ledcWrite(CH_R_LPWM, 0);
}

void stopMotors() {
  ledcWrite(CH_L_RPWM, 0);
  ledcWrite(CH_L_LPWM, 0);
  ledcWrite(CH_R_RPWM, 0);
  ledcWrite(CH_R_LPWM, 0);
}

// ================= LED BLINK (NEW) =================
void blinkLED(int times, int onMs, int offMs) {
  for (int i = 0; i < times; i++) {
    digitalWrite(LED_PIN, HIGH);
    delay(onMs);
    digitalWrite(LED_PIN, LOW);
    delay(offMs);
  }
}

// ================= SETUP =================
void setup() {
  Serial.begin(9600);

  // LED pin setup (NEW)
  pinMode(LED_PIN, OUTPUT);
  digitalWrite(LED_PIN, LOW);

  // Motor Enable Pins
  pinMode(L_REN, OUTPUT);
  pinMode(L_LEN, OUTPUT);
  pinMode(R_REN, OUTPUT);
  pinMode(R_LEN, OUTPUT);

  digitalWrite(L_REN, HIGH);
  digitalWrite(L_LEN, HIGH);
  digitalWrite(R_REN, HIGH);
  digitalWrite(R_LEN, HIGH);

  // PWM Setup
  ledcSetup(CH_L_RPWM, PWM_FREQ, PWM_RES);
  ledcSetup(CH_L_LPWM, PWM_FREQ, PWM_RES);
  ledcSetup(CH_R_RPWM, PWM_FREQ, PWM_RES);
  ledcSetup(CH_R_LPWM, PWM_FREQ, PWM_RES);

  ledcAttachPin(L_RPWM, CH_L_RPWM);
  ledcAttachPin(L_LPWM, CH_L_LPWM);
  ledcAttachPin(R_RPWM, CH_R_RPWM);
  ledcAttachPin(R_LPWM, CH_R_LPWM);

  // IR Sensors
  pinMode(IR_LEFT, INPUT);
  pinMode(IR_RIGHT, INPUT);

  // RFID
  SPI.begin();        // ESP32 default VSPI pins
  mfrc522.PCD_Init();
  Serial.println("RFID Ready");

  // Stepper pins
  pinMode(EN_PIN, OUTPUT);
  digitalWrite(EN_PIN, LOW); // enable A4988 (LOW = enabled)

  stepper.setMaxSpeed(STEPPER_MAX_SPEED);
  stepper.setAcceleration(STEPPER_ACCEL);
  stepper.setCurrentPosition(0);

  stopMotors();
}

// ================= LOOP =================
void loop() {

  // --------- MODE: LINE FOLLOW ---------
  if (mode == LINE_FOLLOW) {

    // RFID check (trigger stepper sequence)
    if (mfrc522.PICC_IsNewCardPresent() && mfrc522.PICC_ReadCardSerial()) {
      Serial.println("RFID detected -> STOP and rotate stepper CW 15 rounds");

      stopMotors();

      // Blink LED when RFID detected (NEW)
      blinkLED(1, 150, 150);

      // Start CW move
      stepper.moveTo(TARGET_STEPS); // go to +TARGET_STEPS
      mode = STEPPER_CW;

      mfrc522.PICC_HaltA();
      mfrc522.PCD_StopCrypto1();
      return;
    }

    // Normal line follow
    int leftIR = digitalRead(IR_LEFT);
    int rightIR = digitalRead(IR_RIGHT);

    if (leftIR == LOW && rightIR == LOW) forward();
    else if (leftIR == LOW && rightIR == HIGH) leftTurn();
    else if (leftIR == HIGH && rightIR == LOW) rightTurn();
    else stopMotors();

    delay(10);
    return;
  }

  // --------- MODE: STEPPER CW ---------
  if (mode == STEPPER_CW) {
    stopMotors();           // robot must not move
    stepper.run();          // keep stepping smoothly

    if (stepper.distanceToGo() == 0) {
      Serial.println("CW done -> rotate CCW 15 rounds");

      // Now go back to 0 (CCW 15 rounds)
      stepper.moveTo(0);
      mode = STEPPER_CCW;
    }
    return;
  }

  // --------- MODE: STEPPER CCW ---------
  if (mode == STEPPER_CCW) {
    stopMotors();
    stepper.run();

    if (stepper.distanceToGo() == 0) {
      Serial.println("CCW done -> resume line follow");
      mode = LINE_FOLLOW;
    }
    return;
  }
}
