// ================= TIME SYNC (TLS) =================
void initTime() {
  configTime(19800, 0, "pool.ntp.org", "time.nist.gov"); // Sri Lanka UTC+5:30
  Serial.print("Syncing time");
  time_t now = time(nullptr);
  int retry = 0;
  while (now < 1700000000 && retry < 60) {
    Serial.print(".");
    delay(500);
    now = time(nullptr);
    retry++;
  }
  Serial.println();
  if (now < 1700000000) Serial.println("⚠️ Time sync failed. TLS may fail.");
}

String nowRFC3339Z() {
  time_t now = time(nullptr);
  if (now < 1700000000) return "1970-01-01T00:00:00Z";

  struct tm tm_utc;
  gmtime_r(&now, &tm_utc);

  char buf[25];
  strftime(buf, sizeof(buf), "%Y-%m-%dT%H:%M:%SZ", &tm_utc);
  return String(buf);
}

// ================= RFID =================
String getRFIDTag() {
  String tagID = "";
  for (byte i = 0; i < mfrc522.uid.size; i++) {
    tagID += String(mfrc522.uid.uidByte[i] < 0x10 ? "0" : "");
    tagID += String(mfrc522.uid.uidByte[i], HEX);
  }
  tagID.toUpperCase();
  return tagID;
}

String makeEventDocId() {
  uint32_t r = (uint32_t)esp_random();
  return String(millis()) + "_" + String(r, HEX);
}

// ================= PUMPS (ON/OFF ONLY) =================
// Because IN2 and IN4 are tied to GND, we only drive IN1 and IN3.
void pump1On()  { digitalWrite(PUMP1_CTRL, HIGH); }
void pump1Off() { digitalWrite(PUMP1_CTRL, LOW);  }

void pump2On()  { digitalWrite(PUMP2_CTRL, HIGH); }
void pump2Off() { digitalWrite(PUMP2_CTRL, LOW);  }

// ------------------------- FIRESTORE WRITE FUNCTIONS -------------------------
void updateRobotDoc() {
  if (!Firebase.ready()) return;

  String documentPath = String(ROBOTS_COLLECTION) + "/" + String(ROBOT_ID);

  FirebaseJson content;
  content.set("fields/robotId/stringValue", String(ROBOT_ID));
  content.set("fields/rfid/stringValue", currentRFID);
  content.set("fields/moving/stringValue", robotMovingStatus);
  content.set("fields/scanning/stringValue", scanningStatus);
  content.set("fields/actuator/stringValue", actuatorStatus);
  content.set("fields/mode/stringValue", modeStr);
  content.set("fields/rssi/integerValue", (int)WiFi.RSSI());
  content.set("fields/updatedAt/timestampValue", nowRFC3339Z());

  String mask = "robotId,rfid,moving,scanning,actuator,mode,rssi,updatedAt";

  if (!Firebase.Firestore.patchDocument(&fbdo, FIREBASE_PROJECT_ID, "",
                                        documentPath.c_str(), content.raw(), mask.c_str())) {
    Serial.print("Robot doc update FAILED: ");
    Serial.println(fbdo.errorReason());
  }
}

// ================= SETUP =================
void setup() {
  Serial.begin(115200);

  // === WIFI ===
  Serial.print("Connecting to WiFi");
  WiFi.begin(WIFI_SSID, WIFI_PASSWORD);
  while (WiFi.status() != WL_CONNECTED) { Serial.print("."); delay(300); }
  Serial.println("\nConnected to WiFi!");
  WiFi.setSleep(false);

  // TLS time
  initTime();

  // === FIREBASE INIT ===
  config.api_key = API_KEY;
  config.token_status_callback = tokenStatusCallback;

  if (Firebase.signUp(&config, &auth, "", "")) {
    Serial.println("Firebase Auth Ready");
  } else {
    Serial.print("Signup error: ");
    Serial.println(config.signer.signupError.message.c_str());
  }

  fbdo.setBSSLBufferSize(4096, 1024);
  fbdo.setResponseSize(4096);

  Firebase.begin(&config, &auth);
  Firebase.reconnectWiFi(true);

  // Pumps control pins
  pinMode(PUMP1_CTRL, OUTPUT);
  pinMode(PUMP2_CTRL, OUTPUT);
  pump1Off();
  pump2Off();

  // Servo
  nozzleServo.setPeriodHertz(50);
  nozzleServo.attach(SERVO_PIN, 500, 2400);
  nozzleServo.write(SERVO_ANGLE_A);

  // Motor Enable Pins
  pinMode(L_REN, OUTPUT);
  pinMode(L_LEN, OUTPUT);
  pinMode(R_REN, OUTPUT);
  pinMode(R_LEN, OUTPUT);

  digitalWrite(L_REN, HIGH);
  digitalWrite(L_LEN, HIGH);
  digitalWrite(R_REN, HIGH);
  digitalWrite(R_LEN, HIGH);

  // PWM Setup
  ledcSetup(CH_L_RPWM, PWM_FREQ, PWM_RES);
  ledcSetup(CH_L_LPWM, PWM_FREQ, PWM_RES);
  ledcSetup(CH_R_RPWM, PWM_FREQ, PWM_RES);
  ledcSetup(CH_R_LPWM, PWM_FREQ, PWM_RES);

  ledcAttachPin(L_RPWM, CH_L_RPWM);
  ledcAttachPin(L_LPWM, CH_L_LPWM);
  ledcAttachPin(R_RPWM, CH_R_RPWM);
  ledcAttachPin(R_LPWM, CH_R_LPWM);

  // IR Sensors
  pinMode(IR_LEFT, INPUT);
  pinMode(IR_RIGHT, INPUT);

  // RFID
  SPI.begin();        // uses ESP32 VSPI default: SCK=18, MISO=19, MOSI=23
  mfrc522.PCD_Init();
  Serial.println("RFID Ready");

  // Stepper enable
  pinMode(EN_PIN, OUTPUT);
  digitalWrite(EN_PIN, LOW);

  stepper.setMaxSpeed(STEPPER_MAX_SPEED);
  stepper.setAcceleration(STEPPER_ACCEL);
  stepper.setCurrentPosition(0);

  stopMotors();

  pushDataToFirebase();
  logScanEvent("BOOT", "Robot booted and connected to WiFi/Firebase");
}

// ================= LOOP =================
void loop() {

  // --------- MODE: LINE FOLLOW ---------
  if (mode == LINE_FOLLOW) {
    modeStr = "LINE_FOLLOW";

    if (mfrc522.PICC_IsNewCardPresent() && mfrc522.PICC_ReadCardSerial()) {
      String readRFID = getRFIDTag();
      unsigned long nowMs = millis();

      // cooldown
      if ((nowMs - lastRFIDTimeMs) < RFID_COOLDOWN_MS && readRFID == lastRFID) {
        mfrc522.PICC_HaltA();
        mfrc522.PCD_StopCrypto1();
        return;
      }
      lastRFID = readRFID;
      lastRFIDTimeMs = nowMs;

      stopMotors();

      currentRFID = readRFID;
      robotMovingStatus = "STOPPED";
      scanningStatus = "SCANNING";
      actuatorStatus = "OFF";

      pushDataToFirebase();
      updatePlantDocOnRFID();
      logScanEvent("RFID_DETECTED", "Robot stopped, scanning started");

      // Start CW
      stepper.moveTo(TARGET_STEPS);
      mode = STEPPER_CW;
      modeStr = "STEPPER_CW";
      logScanEvent("STEPPER_CW_START", "Stepper CW started");

      mfrc522.PICC_HaltA();
      mfrc522.PCD_StopCrypto1();
      return;
    }

    // Line follow
    int leftIR = digitalRead(IR_LEFT);
    int rightIR = digitalRead(IR_RIGHT);

    if (leftIR == LOW && rightIR == LOW) forward();
    else if (leftIR == LOW && rightIR == HIGH) leftTurn();
    else if (leftIR == HIGH && rightIR == LOW) rightTurn();
    else stopMotors();

    delay(10);
    return;
  }

  // --------- MODE: STEPPER CW ---------
  if (mode == STEPPER_CW) {
    modeStr = "STEPPER_CW";
    stopMotors();
    stepper.run();

    if (stepper.distanceToGo() == 0) {
      logScanEvent("STEPPER_CW_DONE", "Stepper CW done");
      stepper.moveTo(0);
      mode = STEPPER_CCW;
      modeStr = "STEPPER_CCW";
      logScanEvent("STEPPER_CCW_START", "Stepper CCW started");
    }
    return;
  }

  // --------- MODE: STEPPER CCW ---------
  if (mode == STEPPER_CCW) {
    modeStr = "STEPPER_CCW";
    stopMotors();
    stepper.run();

    if (stepper.distanceToGo() == 0) {
      logScanEvent("STEPPER_CCW_DONE", "Stepper CCW done, starting pumps+servo");

      scanningStatus = "DONE";
      pushDataToFirebase();

      // Pump1: 5s
      actuatorStatus = "PUMP1_ON";
      pushDataToFirebase();
      logScanEvent("PUMP1_ON", "Pump1 ON for 5s");

      pump1On();
      delay(5000);
      pump1Off();

      actuatorStatus = "OFF";
      pushDataToFirebase();
      logScanEvent("PUMP1_OFF", "Pump1 OFF");

      // Pump2: 5s
      actuatorStatus = "PUMP2_ON";
      pushDataToFirebase();
      logScanEvent("PUMP2_ON", "Pump2 ON for 5s");

      pump2On();
      delay(5000);
      pump2Off();

      actuatorStatus = "OFF";
      pushDataToFirebase();
      logScanEvent("PUMP2_OFF", "Pump2 OFF");

      // Servo: 45 -> 90 (2s) -> 45
      actuatorStatus = "SERVO_MOVING";
      pushDataToFirebase();
      logScanEvent("SERVO_START", "Servo 45->90 wait2s ->45");

      nozzleServo.write(SERVO_ANGLE_A);
      delay(200);
      nozzleServo.write(SERVO_ANGLE_B);
      delay(2000);
      nozzleServo.write(SERVO_ANGLE_A);
      delay(400);

      actuatorStatus = "OFF";
      pushDataToFirebase();
      logScanEvent("SERVO_DONE", "Servo done");

      updatePlantDocAfterAll();
      logScanEvent("PLANT_DONE", "Plant finished (pumps+servo done)");

      robotMovingStatus = "MOVING";
      scanningStatus = "IDLE";
      pushDataToFirebase();
      logScanEvent("RESUME", "Resuming line follow");

      mode = LINE_FOLLOW;
      modeStr = "LINE_FOLLOW";
    }
    return;
  }
