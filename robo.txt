// ================= TIME SYNC (TLS) =================
void initTime() {
  configTime(19800, 0, "pool.ntp.org", "time.nist.gov"); // Sri Lanka UTC+5:30
  Serial.print("Syncing time");
  time_t now = time(nullptr);
  int retry = 0;
  while (now < 1700000000 && retry < 60) {
    Serial.print(".");
    delay(500);
    now = time(nullptr);
    retry++;
  }
  Serial.println();
  if (now < 1700000000) Serial.println("Time sync failed. TLS may fail.");
}

String nowRFC3339Z() {
  time_t now = time(nullptr);
  if (now < 1700000000) return "1970-01-01T00:00:00Z";

  struct tm tm_utc;
  gmtime_r(&now, &tm_utc);

  char buf[25];
  strftime(buf, sizeof(buf), "%Y-%m-%dT%H:%M:%SZ", &tm_utc);
  return String(buf);
}

// ================= RFID =================
String getRFIDTag() {
  String tagID = "";
  for (byte i = 0; i < mfrc522.uid.size; i++) {
    tagID += String(mfrc522.uid.uidByte[i] < 0x10 ? "0" : "");
    tagID += String(mfrc522.uid.uidByte[i], HEX);
  }
  tagID.toUpperCase();
  return tagID;
}

String makeEventDocId() {
  uint32_t r = (uint32_t)esp_random();
  return String(millis()) + "_" + String(r, HEX);
}

// ------------------------- FIRESTORE HELPERS -------------------------
void fetchAssignedTunnelId() {
  if (!Firebase.ready()) return;

  String path = String(ROBOTS_COLLECTION) + "/" + String(ROBOT_ID);

  if (!Firebase.Firestore.getDocument(&fbdo, FIREBASE_PROJECT_ID, "", path.c_str())) {
    Serial.print("fetchAssignedTunnelId getDocument FAILED: ");
    Serial.println(fbdo.errorReason());
    return;
  }

  FirebaseJson json;
  json.setJsonData(fbdo.payload().c_str());

  FirebaseJsonData data;
  json.get(data, "fields/assignedTunnelId/stringValue");

  if (data.success) {
    String v = data.to<String>();
    if (v != assignedTunnelId) {
      assignedTunnelId = v;
      Serial.print("Assigned tunnel updated: ");
      Serial.println(assignedTunnelId);
    }
  }
}

// Reads: tunnels/{assignedTunnelId}/rfidMap/{RFID} -> fields/plantId/stringValue
void resolvePlantDocIdFromRFID() {
  currentPlantDocId = currentRFID; // fallback

  if (!Firebase.ready()) return;
  if (assignedTunnelId.length() == 0) return;
  if (currentRFID == "NONE") return;

  String mapPath = String(TUNNELS_COLLECTION) + "/" + assignedTunnelId + "/" +
                   String(RFIDMAP_SUBCOLLECTION) + "/" + currentRFID;

  if (!Firebase.Firestore.getDocument(&fbdo, FIREBASE_PROJECT_ID, "", mapPath.c_str())) {
    return;
  }

  FirebaseJson json;
  json.setJsonData(fbdo.payload().c_str());
  FirebaseJsonData data;
  json.get(data, "fields/plantId/stringValue");

  if (data.success) {
    currentPlantDocId = data.to<String>();
  }
}

String buildScanEventPath(const String& eventId) {
  if (assignedTunnelId.length() > 0) {
    return String(TUNNELS_COLLECTION) + "/" + assignedTunnelId + "/" +
           String(SCANS_SUBCOLLECTION) + "/" + eventId;

  return String("scanEvents") + "/" + eventId;

String buildPlantPath() {

           String(PLANTS_SUBCOLLECTION) + "/" + currentPlantDocId;

  return String("plants") + "/" + currentRFID;
}

void handleStepperScanMode() {
  modeStr = "STEPPER_SCAN";
  stopMotors();

  if (!stepperPauseActive) {
    stepper.run();

    if (stepper.distanceToGo() == 0) {
      stepperPauseActive = true;
      stepperPauseUntil = millis() + STEPPER_STOP_PAUSE_MS;

      stepperStopIndex++;

      long rounds = stepper.currentPosition() / MOTOR_STEPS_PER_REV;
      String stepTag = currentPositionLabel + String(stepperStopIndex) + " Done";

      Serial.print("Stop reached at ");
      Serial.print(rounds);
      Serial.print(" rounds. Tag: ");
      Serial.println(stepTag);

      String note = stepTag + " - reached " + String(rounds) + " rounds, pause 2s";
      logScanEvent("STEPPER_STOP", note);
    }
    return;
  }

  if (millis() < stepperPauseUntil) {
    return;
  }

  stepperPauseActive = false;

  if (movingForward) {
    if (nextStop < FINAL_TARGET) {
      nextStop += STEPS_PER_STOP;

      Serial.print("Next forward target: ");
      Serial.print(nextStop / MOTOR_STEPS_PER_REV);
      Serial.println(" rounds");

      String note = currentPositionLabel + " next forward target " +
                    String(nextStop / MOTOR_STEPS_PER_REV) + " rounds";
      logScanEvent("STEPPER_NEXT_FORWARD", note);

      stepper.moveTo(nextStop);
      return;
    } else {
      movingForward = false;
      nextStop -= STEPS_PER_STOP;

      Serial.println("50 rounds reached. Switching to CCW.");
      logScanEvent("STEPPER_SWITCH_CCW", currentPositionLabel + " reached 50 rounds, switching to CCW");

      stepper.moveTo(nextStop);
      return;
    }
  } else {
    if (nextStop > 0) {
      nextStop -= STEPS_PER_STOP;

      Serial.print("Next backward target: ");
      Serial.print(nextStop / MOTOR_STEPS_PER_REV);
      Serial.println(" rounds");

      String note = currentPositionLabel + " next backward target " +
                    String(nextStop / MOTOR_STEPS_PER_REV) + " rounds";
      logScanEvent("STEPPER_NEXT_BACKWARD", note);

      stepper.moveTo(nextStop);
      return;
    } else {
      Serial.println("Stepper back at 0. Scan complete.");
      logScanEvent("STEPPER_BACK_AT_ZERO", currentPositionLabel + " stepper returned to 0");

      runActuationSequence();
      return;
    }
  }
}

void handlePositionStop() {
  plantStopCount++;
  currentPositionLabel = getPositionLabelFromCount(plantStopCount);

  stopMotors();
  robotMovingStatus = "STOPPED";
  scanningStatus = "WAITING";
  modeStr = "LINE_FOLLOW";
  pushDataToFirebase();

  Serial.println(">>> [STOP] POSITION " + currentPositionLabel + " DETECTED");
  Serial.println(">>> Waiting 5 seconds...");
  logScanEvent("POSITION_DETECTED", "Position " + currentPositionLabel + " detected, waiting 5s");

  delay(5000);

  robotMovingStatus = "STOPPED";
  scanningStatus = "SCANNING";
  pushDataToFirebase();

  startStepperScanSequence();
}

// ================= SETUP =================
void setup() {
  Serial.begin(115200);

  // === WIFI ===
  Serial.print("Connecting to WiFi");
  WiFi.begin(WIFI_SSID, WIFI_PASSWORD);
  while (WiFi.status() != WL_CONNECTED) {
    Serial.print(".");
    delay(300);
  }
  Serial.println("\nConnected to WiFi!");
  WiFi.setSleep(false);

  // TLS time
  initTime();

  // === FIREBASE INIT ===
  config.api_key = API_KEY;
  config.token_status_callback = tokenStatusCallback;

  if (Firebase.signUp(&config, &auth, "", "")) {
    Serial.println("Firebase Auth Ready");
  } else {
    Serial.print("Signup error: ");
    Serial.println(config.signer.signupError.message.c_str());
  }

  fbdo.setBSSLBufferSize(4096, 1024);
  fbdo.setResponseSize(4096);

  Firebase.begin(&config, &auth);
  Firebase.reconnectWiFi(true);

  // Pumps control pins
  pinMode(PUMP1_CTRL, OUTPUT);
  pinMode(PUMP2_CTRL, OUTPUT);
  pump1Off();
  pump2Off();

  // Servo
nozzleServo.setPeriodHertz(50);

  // Motor Enable Pins (only if actually controlled by ESP32)
  if (USE_GPIO_EN_PINS) {
    pinMode(L_REN, OUTPUT);
    pinMode(L_LEN, OUTPUT);
    pinMode(R_REN, OUTPUT);
    pinMode(R_LEN, OUTPUT);

    digitalWrite(L_REN, HIGH);
    digitalWrite(L_LEN, HIGH);
    digitalWrite(R_REN, HIGH);
    digitalWrite(R_LEN, HIGH);
  }

  // PWM Setup
  ledcSetup(CH_L_RPWM, PWM_FREQ, PWM_RES);
  ledcSetup(CH_L_LPWM, PWM_FREQ, PWM_RES);
  ledcSetup(CH_R_RPWM, PWM_FREQ, PWM_RES);
  ledcSetup(CH_R_LPWM, PWM_FREQ, PWM_RES);

  ledcAttachPin(L_RPWM, CH_L_RPWM);
  ledcAttachPin(L_LPWM, CH_L_LPWM);
  ledcAttachPin(R_RPWM, CH_R_RPWM);
  ledcAttachPin(R_LPWM, CH_R_LPWM);

  // IR Sensors
  pinMode(IR_LEFT, INPUT);
  pinMode(IR_RIGHT, INPUT);

  // RFID
  SPI.begin();
  mfrc522.PCD_Init();
  Serial.println("RFID Ready");

  // Stepper enable
  pinMode(EN_PIN, OUTPUT);
  digitalWrite(EN_PIN, LOW);

  stepper.setMaxSpeed(STEPPER_MAX_SPEED);
  stepper.setAcceleration(STEPPER_ACCEL);
  stepper.setCurrentPosition(0);

  stopMotors();

  // Load assigned tunnel once on boot
  fetchAssignedTunnelId();
  lastTunnelFetchMs = millis();

  pushDataToFirebase();
  logScanEvent("BOOT", "Robot booted and connected to WiFi/Firebase");

  Serial.println(">>> SYSTEM READY");
  Serial.println(">>> Robot starts with line following");
  Serial.println(">>> After RFID detected, next both-black = A, then B, then C");
}

// ================= LOOP =================
void loop() {
  if (Firebase.ready()) {
    unsigned long nowMs = millis();
    if (nowMs - lastTunnelFetchMs > TUNNEL_REFRESH_MS) {
      fetchAssignedTunnelId();
      lastTunnelFetchMs = nowMs;
    }
  }

  if (mode == FINISHED) {
    stopMotors();
    return;
  }

  if (mode == STEPPER_SCAN) {
    handleStepperScanMode();
    return;
  }

  if (mode == ACTUATION) {
    return;
  }

  // --------- MODE: LINE FOLLOW ---------
  if (mode == LINE_FOLLOW) {
    modeStr = "LINE_FOLLOW";

    int leftIR = digitalRead(IR_LEFT);
    int rightIR = digitalRead(IR_RIGHT);

    bool bothBlack = (leftIR == HIGH && rightIR == HIGH);

    // ===== RFID DETECTION =====
    if (!rfidDetected) {
      if (mfrc522.PICC_IsNewCardPresent() && mfrc522.PICC_ReadCardSerial()) {
        String readRFID = getRFIDTag();
        unsigned long nowMs = millis();

        if ((nowMs - lastRFIDTimeMs) >= RFID_COOLDOWN_MS || readRFID != lastRFID) {
          lastRFID = readRFID;
          lastRFIDTimeMs = nowMs;

          rfidDetected = true;
          plantStopCount = 0;
          currentPositionLabel = "RFID";
          currentRFID = readRFID;
          robotMovingStatus = "STOPPED";
          scanningStatus = "RFID_DETECTED";
          actuatorStatus = "OFF";

          stopMotors();
          pushDataToFirebase();
          updatePlantDocOnRFID();
          logScanEvent("RFID_DETECTED", "RFID detected. Next both-black will be position A");

          Serial.println(">>> [RFID] TAG DETECTED");
          Serial.println(">>> Next both-black will be POSITION A");

          delay(1000);
        }

        mfrc522.PICC_HaltA();
        mfrc522.PCD_StopCrypto1();
      }
    }

    // ===== POSITION DETECTION AFTER RFID =====
    if (rfidDetected) {
      if (bothBlack && !bothBlackLatched) {
        bothBlackLatched = true;
        handlePositionStop();
        return;
      }

      if (!bothBlack) {
        bothBlackLatched = false;
      }
    }

    // ===== NORMAL LINE FOLLOWING =====
    if (leftIR == LOW && rightIR == HIGH) {
      leftTurn();
    }
    else if (leftIR == HIGH && rightIR == LOW) {
      rightTurn();
    }
    else {
      forward();
    }

    robotMovingStatus = "MOVING";
    scanningStatus = "IDLE";
    actuatorStatus = "OFF";

    delay(10);
    return;
  }
}
